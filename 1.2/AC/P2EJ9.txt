CANT DB 4
MSJ1 DB "ACCESO PERMITIDO"
MSJ2 DB "ACCESO DENEGADO"
FINM DB ?

ORG 1500H
NUM DB ?

ORG 2000H
MOV DX, 0
MOV CX, 0
LUUP: MOV BX, OFFSET NUM
INT 6
MOV AL, [BX]
MOV BX, OFFSET CLAVE
ADD BX, CX
CMP AL, [BX]
JZ BIEN
INC CX
CMP CX, 4
JZ DENEGAR
JMP LUUP

BIEN: INC DX
CMP DX, 4
JZ PERMITIR
INC CX
CMP CX, 4
JZ FIN
JMP LUUP

PERMITIR: MOV BX, OFFSET MSJ1
MOV AL, OFFSET MSJ2 - OFFSET MSJ1
INT 7
JMP FIN

DENEGAR: MOV BX, OFFSET MSJ2
MOV AL, OFFSET FINM - OFFSET MSJ2
INT 7
JMP FIN

FIN: INT 0
END
